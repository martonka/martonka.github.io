00005 // A Status encapsulates the result of an operation.  It may indicate success,
00006 // or it may indicate an error with an associated error message.
00007 //
00008 // Multiple threads can invoke const methods on a Status without
00009 // external synchronization, but if any of the threads may call a
00010 // non-const method, all threads accessing the same Status must use
00011 // external synchronization.

00379 inline void Status::operator=(const Status& s) {
00380   // The following condition catches both aliasing (when this == &s),
00381   // and the common case where both s and *this are ok.
