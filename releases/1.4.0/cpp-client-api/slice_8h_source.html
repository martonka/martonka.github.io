284 inline int Slice::compare(const Slice& b) const {
285  const int min_len = (size_ < b.size_) ? size_ : b.size_;
286  int r = MemCompare(data_, b.data_, min_len);
287  if (r == 0) {
288  if (size_ < b.size_) r = -1;
289  else if (size_ > b.size_) r = +1;
290  }
291  return r;
292 }
